function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("DATA PT"); var p = e.parameter; // Ghi đúng 8 cột: Date, Time, WH, Team, Vendor, WFM ID, Name, Email sheet.appendRow([ p.date, p.time, p.wh, p.team, p.vendor, p.wfmId, p.name, p.email ]); return ContentService.createTextOutput("Success").setMimeType(ContentService.MimeType.TEXT); } function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("DATA PT"); var data = sheet.getDataRange().getValues(); var result = []; for (var i = 1; i < data.length; i++) { if (data[i][0]) { // Ép kiểu Ngày yyyy-MM-dd để bộ lọc HTML tìm được var d = new Date(data[i][0]); var fDate = d.getFullYear() + "-" + ("0" + (d.getMonth() + 1)).slice(-2) + "-" + ("0" + d.getDate()).slice(-2); // Ép kiểu Giờ HH:mm (Xử lý lỗi 1899) var rawT = data[i][1]; var fTime = ""; if (rawT instanceof Date) { fTime = ("0" + rawT.getHours()).slice(-2) + ":" + ("0" + rawT.getMinutes()).slice(-2); } else { fTime = rawT.toString().substring(0, 5); } result.push({ date: fDate, time: fTime, wh: data[i][2], team: data[i][3], vendor: data[i][4], wfmId: data[i][5], name: data[i][6] }); } } return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); }
